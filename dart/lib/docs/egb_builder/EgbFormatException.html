String toString() {
  StringBuffer strBuf = new StringBuffer();
  strBuf.add("Format exception");
  if (line != null) {
    strBuf.add(" on line [$line]");
  }
  if (file != null) {
    strBuf.add(" in file ${file.name}");
  }
  strBuf.add(": ");
  strBuf.add(msg);
  return strBuf.toString();
}
